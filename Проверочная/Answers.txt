1. SELECT *
FROM Водители 
WHERE Стаж NOT IN ('3', '5', '6', '10') AND Место прописки = 'eee' 

2. SELECT Водители.№ паспорта
FROM Водители
LEFT JOIN Диспетчерская ON Водители.Код водителя=Диспетчерская.Код водителя
WHERE Диспетчерская.Код водителя IS NULL

3.SELECT Автомобили.Регистрационный № авто, Диспетчерская.Время прибытия, Диспетчерская.Время отбытия, Маршрут.Пункт назначения
FROM Автомобили
LEFT JOIN Диспетчерская ON Автомобили.Код авто=Диспетчерская.Код авто
LEFT JOIN Маршрут ON Диспетчерская.Путевка=Маршрут.Пункт назначения 
WHERE Автомобили.Год выпуска авто = '2015'
